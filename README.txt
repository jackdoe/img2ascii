copied from https://github.com/stdupp/goasciiart, but made into library

example:

go get https://github.com/jackdoe/img2ascii


usage:

// from image object
img := image.Decode...
ascii := string(Img2Ascii(ScaleImage(img, 80)))

// or from file
ascii := string(MustFile2Ascii(os.Argv[0], 80))


example: 

package main

import (
	"fmt"
	"os"

	"github.com/jackdoe/img2ascii"
)

func main() {
	ascii := string(img2ascii.MustFile2Ascii(os.Args[1], 80))
	fmt.Printf("%s", ascii)
}
